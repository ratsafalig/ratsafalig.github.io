<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="k">auto</span> <span class="n">a</span> <span class="o">=</span> <span class="mi">123</span><span class="p">;</span>

<h2 id="constexpr">constexpr</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">对一个变量始加</span><span class="k">constexpr</span><span class="err">表示该变量将在编译时进行计算</span>

<span class="k">constexpr</span> <span class="kt">int</span> <span class="n">j</span> <span class="o">=</span> <span class="n">i</span> <span class="o">+</span> <span class="mi">123</span><span class="p">;</span><span class="c1">//编译时计算j</span>

<span class="err">如果被始加</span><span class="k">constexpr</span><span class="err">的是一个指针</span><span class="p">,</span><span class="err">那么</span><span class="k">constexpr</span><span class="err">只作为顶层</span><span class="k">const</span><span class="err">起作用进行强制计算</span>

<h2 id="decltype">decltype</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="k">decltype</span><span class="err">取得一个表达式的类型</span>
<span class="kt">int</span> <span class="n">i</span> <span class="o">=</span> <span class="mi">123</span><span class="p">;</span>
<span class="k">decltype</span><span class="p">(</span><span class="n">i</span> <span class="o">+</span> <span class="mi">0</span><span class="p">)</span> <span class="n">j</span><span class="p">;</span>

<h2 id="explicit">explicit</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">默认情况下如果存在一个参数的构造</span><span class="p">,</span><span class="n">C</span><span class="o">++</span><span class="err">允许隐式的类型转换</span><span class="p">.</span>

<span class="p">}</span>
<span class="err">这种情况下可以自动类型转换</span><span class="o">:</span>

<span class="err">可以通过将一个参数的构造声明成</span><span class="n">explict</span><span class="err">来取消这种隐式转换</span><span class="o">:</span>

    <span class="k">explicit</span> <span class="n">A</span><span class="p">(</span><span class="kt">int</span> <span class="n">i</span><span class="p">){...}</span>

<span class="err">这种情形下就只能利用</span><span class="k">new</span><span class="err">来显示的转换</span><span class="o">:</span>

        <span class="k">return</span> <span class="mi">123</span><span class="p">;</span>
    <span class="p">}</span>
<span class="p">}</span>
<span class="err">这样的话允许出现这样操作</span><span class="o">:</span>

<span class="n">a</span> <span class="o">=</span> <span class="n">a</span> <span class="o">+</span> <span class="mi">3</span><span class="p">;</span>
<span class="n">C</span><span class="o">++</span><span class="err">会隐式的把</span><span class="n">a</span><span class="err">转换成</span><span class="kt">int</span><span class="p">,</span>

    <span class="k">explicit</span> <span class="k">operator</span> <span class="kt">int</span><span class="p">()</span> <span class="k">const</span><span class="p">{</span>
        <span class="k">return</span> <span class="mi">123</span><span class="p">;</span>
    <span class="p">}</span>
<span class="p">}</span>

<h2 id="final">final</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">类似</span><span class="n">java</span><span class="p">,</span><span class="n">C</span><span class="o">++</span><span class="err">新版本引入</span><span class="k">final</span><span class="err">和</span><span class="k">override</span>

    <span class="k">virtual</span> <span class="kt">void</span> <span class="n">_2</span><span class="p">(){...}</span> 
    <span class="kt">void</span> <span class="n">_3</span><span class="p">(){...}</span>
<span class="p">}</span>

<h2 id="inline">inline</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">假设下面函数</span><span class="o">:</span>

<span class="err">因为函数实在太简单了</span><span class="p">,</span><span class="err">以至于可以用一行表达式写出来</span><span class="p">(</span><span class="err">直接计算</span><span class="n">a</span><span class="o">+</span><span class="n">b</span><span class="p">),</span>

<span class="err">实际上</span><span class="k">constexpr</span><span class="err">也有差不多的效果</span>
<span class="k">constexpr</span> <span class="kt">int</span> <span class="nf">func</span><span class="p">(</span><span class="kt">int</span> <span class="n">a</span><span class="p">,</span><span class="kt">int</span> <span class="n">b</span><span class="p">){</span><span class="k">return</span> <span class="n">a</span> <span class="o">+</span> <span class="n">b</span><span class="p">;};</span>
<span class="err">这样的话就会在编译时直接计算</span><span class="n">a</span> <span class="o">+</span> <span class="n">b</span><span class="p">,</span><span class="err">编译器其实是把</span><span class="k">constexpr</span><span class="err">函数先转成</span><span class="kr">inline</span><span class="p">,</span>

<h2 id="lambda">lambda</h2>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="n">lambda</span><span class="err">是一种函数式编程的工具</span><span class="p">,</span><span class="err">形式如下</span><span class="o">:</span>

<span class="err">捕获列表可以允许捕获本地栈变量</span><span class="o">:</span>

    <span class="kt">int</span> <span class="n">i</span> <span class="o">=</span> <span class="mi">123</span><span class="p">;</span>

<span class="mf">3.</span> <span class="p">[</span><span class="o">&amp;</span><span class="p">],</span><span class="err">隐式捕获列表</span><span class="p">,</span><span class="err">采用引用捕获方式</span>
<span class="mf">4.</span> <span class="p">[</span><span class="o">=</span><span class="p">],</span><span class="err">隐式捕获列表</span><span class="p">,</span><span class="err">采用值捕获方式</span>
<span class="mf">5.</span> <span class="p">[</span><span class="o">=</span><span class="p">,</span><span class="n">i</span><span class="p">,</span><span class="n">j</span><span class="p">,</span><span class="n">k</span><span class="p">],</span><span class="err">捕获</span><span class="n">i</span><span class="p">,</span><span class="n">j</span><span class="p">,</span><span class="n">k</span><span class="err">变量</span><span class="p">,</span><span class="err">采用值捕获方式</span>
<span class="mf">6.</span> <span class="p">[</span><span class="o">&amp;</span><span class="p">,</span><span class="n">i</span><span class="p">,</span><span class="n">j</span><span class="p">,</span><span class="n">k</span><span class="p">],</span><span class="err">捕获</span><span class="n">i</span><span class="p">,</span><span class="n">j</span><span class="p">,</span><span class="n">k</span><span class="err">变量</span><span class="p">,</span><span class="err">采用引用捕获方式</span>

<span class="err">如果采用值捕获方式</span><span class="p">,</span><span class="n">lambda</span><span class="err">方法体其实是不能改变的</span><span class="p">,</span>
<span class="err">这个原因是因为带捕获列表的</span><span class="n">lambda</span><span class="err">实际上是被转换成一个类的</span><span class="o">:</span>

<span class="k">auto</span> <span class="n">j</span> <span class="o">=</span> <span class="p">[]{</span><span class="k">return</span> <span class="mi">0</span><span class="p">;};</span>
<span class="err">这种情况下</span><span class="n">j</span><span class="err">是一个函数指针</span><span class="o">:</span>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">除了函数指针</span><span class="p">,</span><span class="n">C</span><span class="o">++</span><span class="err">提供了另一种接收</span><span class="n">Lambda</span><span class="err">的方式</span><span class="p">,</span>

<h3 id="shared_ptr">shared_ptr&lt;&gt;</h3>

<div class="language-cpp highlighter-rouge"><div class="highlight"><pre class="highlight"><code><span class="err">最常用的智能指针</span><span class="p">,</span><span class="err">采用引用计数方式</span><span class="p">,</span><span class="err">如果当前对象没有被任何人引用</span><span class="p">,</span><span class="err">则释放</span><span class="o">:</span>

<span class="n">shared_ptr</span><span class="o">&lt;</span><span class="n">A</span><span class="o">&gt;</span> <span class="n">l</span><span class="p">(</span><span class="k">new</span> <span class="nf">A</span><span class="p">(</span><span class="mi">123</span><span class="p">));</span>
